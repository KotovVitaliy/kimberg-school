CREATE TABLE  `u483211`.`accounts` (
`id` INT NOT NULL AUTO_INCREMENT PRIMARY KEY ,
`name` varchar (150) NOT NULL,
`fname` varchar (150) NOT NULL,
`role` enum ('teacher', 'student', 'parent') DEFAULT 'student' ,
`email` VARCHAR( 100 ) NOT NULL ,
`password` VARCHAR( 50 ) NOT NULL ,
`vk_id` VARCHAR( 100 ) DEFAULT NULL,
`inst_id` VARCHAR( 100 ) DEFAULT NULL,
`photo_id` INT ( 32 ) DEFAULT NULL,
`current_class` enum ('7', '8', '9', '10', '11', '0') DEFAULT '0',
`created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
`confirm_token` VARCHAR( 50 ) NOT NULL ,
`forgot_token` VARCHAR( 50 ) NOT NULL DEFAULT '0' ,
UNIQUE (`email`),
INDEX email_password (`email`, `password`),
INDEX user_id_confirm_token (`id`, `confirm_token`)
) ENGINE = 'InnoDB' CHARACTER SET utf8 COLLATE utf8_general_ci;

CREATE TABLE `authorizer` (
 `user_id` int(11) NOT NULL,
 `hash` varchar(40) NOT NULL,
 `created` TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP ,
) ENGINE = 'MyISAM' CHARACTER SET utf8 COLLATE utf8_general_ci;
